假设我们要编写一个地图应用，现在有两家可选的地图API提供商供我们接入自己的应用。目前我们选择的是百度地图，百度地图的API中提供了show方法，负责在页面上展示整个地图。示例代码如下：

				<script type="text/javascript">
					//地图对象
					var baiduMap={
						show:function(){
							console.log('正在加载渲染百度地图.');
						}
					};
					var rendMap=function(){
						baiduMap.show();
					};
					rendMap();//输出：正在加载渲染百度地图.
				</script>

现在想换成谷歌地图，则修改代码如下

				<script type="text/javascript">
					//百度地图对象
					var baiduMap={
						show:function(){
							console.log('正在加载渲染百度地图.');
						}
					};
					//谷歌地图
					var googleMap={
						show:function(){
							console.log('正在加载渲染谷歌地图.');
						}
					};
					var rendMap=function(type){
						if(type=='google'){
							googleMap.show();
						}else if(type=='baidu'){
							baiduMap.show();
						}
					};
					rendMap('baidu');//输出：正在加载渲染百度地图.
					rendMap('google');//输出：正在加载渲染谷歌地图.
				</script>

如上情况，假设我再次添加一些其他地图，则又要修改rendMap方法，因此我们要进行抽离不变和可变的进行封装处理。见JavaScript王国--地图2
