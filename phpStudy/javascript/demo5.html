JavaScript王国--地图2

根据demo4上面的总结，我们还是先把程序中相同的部分抽象出来，那就是显示某个地图:

				<script type="text/javascript">
					//百度
					var baiduMap={
						show:function(){
							console.log('正在加载渲染百度地图.');
						}
					};
					//谷歌
					var googleMap={
						show:function(){
							console.log('正在加载渲染谷歌地图.');
						}
					};
					var renderMap=function(map){
						if(map.show instanceof Function){//判断是否存在show方法，存在则该地图执行该方法
							map.show();
						}
					};
					//使用地图
					renderMap(googleMap);
					renderMap(baiduMap);
					//现在新增地图
					var sosoMap={
						show:function(){
							console.log('正在加载渲染搜搜地图');
						}
					};
					renderMap(sosoMap);
				</script>

当然实际上不可能所有地图都是叫做show方法的，以上均是学习中假设情况。下节开始讲述JavaScript的封装
